命名路由

const routes = [
  {
    path: '/user/:id',
    name: 'user',	// 对路由进行命名
    component: () => import(/* webpackChunkName: "user" */ '../views/User.vue')
  },
]

# 携带参数

this.$router.push({
  name:'user',
  params: {
    id:123
  }
})

# 获取参数

this.$route.params.id
