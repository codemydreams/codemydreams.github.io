# 安装脚手架

vue cliopen in new window

# 安装vue cli脚手架
npm install -g @vue/cli
# 升级vue cli脚手架版本
npm update -g @vue/cli

#查看vue版本
vue --version
vue -V

# 命令行创建项目
vue create project_name
vue create . # 当前目录下创建
# 可视化创建项目
vue ui
